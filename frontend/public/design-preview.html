// Submit an APK for analysis
const res = await fetch("https://api.prodscope.io/v1/analyze", {
  method: "POST",
  headers: { Authorization: `Bearer ${token}` },
  body: JSON.stringify({
    apkUrl: "https://cdn.example.com/app.apk",
    webhook: "https://ci.example.com/hooks/prodscope"
  })
})
const { reportId } = await res.json()
